decrypted_output = []

	plain_byte = int(plain_text[i], 16)
	key_byte = int(key[i], 16)

	result = xor_result
	if result > 0xFF:
		result = result % 0x100
	hexa = hex(result)[2:]

	if len(hexa) == 1:
		hexa = '0' + hexa
	
	decrypted_output.append(hexa)

hex_string = "".join(encrypted_output)
byte_values = bytes.fromhex(hex_string)
